# Problem 3 (revised)
import numpy as np
from math import *
import matplotlib.pyplot as plt
%matplotlib inline

g=9.8
m=0.1
l=g
th0=0.201
q=0.5
FD=1.2
wD=2.0/3.0
n=100          # the number of periods
N1=5000        # iteration steps in each period
Nt=N1*n        # total iteration steps
T=2*pi*n/wD; print('total time=',T,'period=',2*pi/wD)  # total time
dt=T/Nt        # step-length
thth=np.zeros(n+1)  # the angle at the begining of each period
th=th0
ww=np.zeros(n+1)   # the angular velocity at the begining of each period
w=0               # initial angular velocity
w1=-sin(th)       # initial angular acceleration
for ii in range(0,Nt+1):
    if ii%N1==0:          # at the beginning of each period
        thth[ii//N1]=th
        ww[ii//N1]=w
    #more accurate Euler method
    w1=((-sin(th)-q*w+FD*sin(wD*T/Nt*ii))+(-sin(th+w*dt)-q*(w+w1*dt)+FD*sin(wD*T/Nt*(ii+1))))/2 
    th=th+w*dt
    w=w+w1*dt

plt.scatter(ww,thth,color='blue')

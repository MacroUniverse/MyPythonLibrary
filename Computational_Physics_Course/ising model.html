# Total Energy of a given state of the Ising Model
def IsingEnergy(ss,N1,N2):
    # Total Energy
    TotalEnergy = 0
    for ii in range(0,N1):
        for jj in range(0,N2):
                TotalEnergy -= ss[ii,jj] * (ss[ii-1,jj] + ss[ii,jj-1])
    return TotalEnergy

# Generate the initial State for the Ising Model
def IsingInit(N1,N2):
    ss = np.ones([N1,N2],dtype=int); mask = ss>0.5;
    ss[mask] = 1;
    ss[np.logical_not(mask)] = -1;
    ss = ss.astype(int)
    #print(ss)
    return ss

# Flip each spin for each time step according to the rules
def flipping(ss,N1,N2,N,T):
    TotalEnergy = np.zeros(N+1,dtype = int)
    Magnetization = np.zeros(N+1,dtype = int)
    TotalEnergy[0] = IsingEnergy(ss,N1,N2)
    # flipping
    for kk in range(0,N):
        for ii in range(0,N1):
            for jj in range(0,N2):
                if (ii < N1-1) and (jj < N2-1) :
                    dE = 2*ss[ii,jj] * (ss[ii-1,jj] + ss[ii+1,jj] \
                                    + ss[ii,jj-1] + ss[ii,jj+1]);
                else: 
                    dE = 2*ss[ii,jj] * (ss[ii-1,jj] + ss[ii,jj-1])
                    if ii == N1-1: dE += 2*ss[ii,jj] * ss[0,jj]
                    else: dE += 2*ss[ii,jj] * ss[ii+1,jj]
                    if jj == N2-1: dE += 2*ss[ii,jj] * ss[ii,0]
                    else: dE += 2*ss[ii,jj] * ss[ii,jj+1]
                #print('ii=',ii,' jj=',jj)
                #print('dE=',dE,' exp(-dE/T)=', exp(-dE/T))
                if (dE <= 0)  or  (np.random.rand(1) < exp(-dE/T)):
                    #print('flip!')
                    ss[ii,jj] = -ss[ii,jj]
                    
        TotalEnergy[kk+1] = IsingEnergy(ss,N1,N2)
        Magnetization[kk+1] = np.sum(ss)
    return TotalEnergy, Magnetization

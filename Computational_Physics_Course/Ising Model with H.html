# Total Energy of a given state of the Ising Model
def IsingEnergy(H,ss,N1,N2):
    # Total Energy
    TotalEnergy = 0
    for ii in range(0,N1):
        for jj in range(0,N2):
                TotalEnergy -= ss[ii,jj] * (ss[ii-1,jj] + ss[ii,jj-1] + H)
    return TotalEnergy

# Flip each spin for each time step according to the rules
def flipping(ss,N1,N2,N,T,H):
    TotalEnergy = np.zeros(N+1,dtype = int)
    Magnetization = np.zeros(N+1,dtype = int)
    TotalEnergy[0] = IsingEnergy(H,ss,N1,N2)
    Magnetization[0] = np.sum(ss)
    # flipping
    for kk in range(0,N):
        for ii in range(0,N1):
            for jj in range(0,N2):
                if (ii < N1-1) and (jj < N2-1) :
                    dE = 2*ss[ii,jj] * (ss[ii-1,jj] + ss[ii+1,jj] \
                                    + ss[ii,jj-1] + ss[ii,jj+1] + H);
                else: 
                    dE = 2*ss[ii,jj] * (ss[ii-1,jj] + ss[ii,jj-1] + H)
                    if ii == N1-1: dE += 2*ss[ii,jj] * ss[0,jj]
                    else: dE += 2*ss[ii,jj] * ss[ii+1,jj]
                    if jj == N2-1: dE += 2*ss[ii,jj] * ss[ii,0]
                    else: dE += 2*ss[ii,jj] * ss[ii,jj+1]
                #print('ii=',ii,' jj=',jj)
                #print('dE=',dE,' exp(-dE/T)=', exp(-dE/T))
                if (dE <= 0)  or  (np.random.rand(1) < exp(-dE/T)):
                    #print('flip!')
                    ss[ii,jj] = -ss[ii,jj]
                    
        TotalEnergy[kk+1] = IsingEnergy(H,ss,N1,N2)
        Magnetization[kk+1] = np.sum(ss)
    return TotalEnergy, Magnetization

# Critical Temperature
# I don't think in this case there will be any critical temperature.
# Because a spin with its neighbor pointing down will have a total energy of -4+5=1
#      When it change to point up with neighbors still pointing down, total energy
#      will be 4-5=-1. So it's always certain that a down pointing spin will go up next iteration
#      so, the all-down system is not stable, so by definition there will be no
#      critical temperature.

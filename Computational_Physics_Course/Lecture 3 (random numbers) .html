#Calculate the area of a circle

import math
import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

Nin=np.zeros(100)
N=np.zeros(100);
for jj in range(0,30):
    N[jj]=(jj+1)*10000;
    x=np.random.rand(N[jj])*2-1
    y=np.random.rand(N[jj])*2-1
    count=0;
    for ii in range(0,(jj+1)*10000):
        if math.sqrt(x[ii]**2+y[ii]**2)<1 :
            count=count+1
    Nin[jj]=count
plt.plot(N,Nin/N/(math.pi/4)-1)

# Calculate The integral of 1/x from a to b using random number

a=1;b=2
def f(x):
    return 1/x
Nin=np.zeros(100)
N=np.zeros(100);
for jj in range(0,30):
    N[jj]=(jj+1)*30000;
    x=np.random.rand(N[jj])*(b-a)+a
    y=np.random.rand(N[jj])*f(a)
    count=0;
    for ii in range(0,(jj+1)*30000):
        if y[ii] < f(x[ii]) :
            count=count+1
    Nin[jj]=count
plt.plot(N,Nin/N/(math.log(b)-math.log(a))-1)

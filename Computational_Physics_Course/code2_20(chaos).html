# Problem 1(a)
g=9.8
m=1
l=g
th0=0.1
q=0
FD=0
wD=0
Nt=10000
T=10.0;
t=np.linspace(0,T,Nt)
E=np.zeros(Nt)
dt=T/(Nt-1)
th=np.zeros(Nt)
th[0]=th0
w=0
for ii in range(0,Nt-1):
    E[ii]=m*g*l*(1-cos(th[ii]))+1.0/2.0*m*(l**2)*(w**2)
    w1=-sin(th[ii])-q*w+FD*sin(wD*t[ii])
    w=w+w1*dt
    th[ii+1]=th[ii]+w*dt
E[Nt-1]=E[Nt-2]
plt.plot(t,th,color='blue')
plt.plot(t,E,color='red')

# Problem 5 (phase diagram)
import numpy as np
from math import *
import matplotlib.pyplot as plt
%matplotlib inline

g=9.8
m=0.1
l=g
th0=0.201
q=0.5
FD=1.2
wD=2.0/3.0
n=100
N1=5000
Nt=N1*n
T=2*pi*n/wD; print('total time=',T,'period=',2*pi/wD)
dt=T/Nt
thth=np.zeros(n+1)
th=th0
ww=np.zeros(n+1)
w=0
w1=-sin(th)
for ii in range(0,Nt+1):
    if ii%N1==0:
        #print('time=',T/Nt*ii,'sin=',sin(wD*T/Nt*ii))
        #print(ii//N1)
        #print(T/Nt*ii,w,th)
        thth[ii//N1]=th
        ww[ii//N1]=w
    #more accurate Euler method
    w1=((-sin(th)-q*w+FD*sin(wD*T/Nt*ii))+(-sin(th+w*dt)-q*(w+w1*dt)+FD*sin(wD*T/Nt*(ii+1))))/2 
    th=th+w*dt
    w=w+w1*dt

plt.scatter(ww,thth,color='blue')

def main1(M,T,Nt,x0,y0,vx0,vy0):
    
    dt=T/(Nt-1); dt2=dt/2; dt6=dt/6
    x=np.zeros(Nt); y=np.zeros(Nt); vx=np.zeros(Nt); vy=np.zeros(Nt);
    x[0]=x0; y[0]=y0; vx[0]=vx0; vy[0]=vy0;

    def D(x,y,vx,vy):
        Dx=vx; Dy=vy;
        r=sqrt(x**2+y**2);
        Dvx=-M*x/r**3; Dvy=-M*y/r**3;
        return Dx,Dy,Dvx,Dvy

    for ii in range(0,Nt-1):
        k1x,k1y,k1vx,k1vy=D(x[ii],y[ii],vx[ii],vy[ii])
        k2x,k2y,k2vx,k2vy=D(x[ii]+dt2*k1x,y[ii]+dt2*k1y,vx[ii]+dt2*k1vx,vy[ii]+dt2*k1vy)
        k3x,k3y,k3vx,k3vy=D(x[ii]+dt2*k2x,y[ii]+dt2*k2y,vx[ii]+dt2*k2vx,vy[ii]+dt2*k2vy)
        k4x,k4y,k4vx,k4vy=D(x[ii]+dt *k3x,y[ii]+dt *k3y,vx[ii]+dt *k3vx,vy[ii]+dt *k3vy)
        x[ii+1]=x[ii]+dt6*(k1x+2*k2x+2*k3x+k4x)
        y[ii+1]=y[ii]+dt6*(k1y+2*k2y+2*k3y+k4y)
        vx[ii+1]=vx[ii]+dt6*(k1vx+2*k2vx+2*k3vx+k4vx)
        vy[ii+1]=vy[ii]+dt6*(k1vy+2*k2vy+2*k3vy+k4vy)
    t=np.linspace(0,T,Nt);
    plt.figure()
    plt.plot(t,0.5*(vx**2+vy**2));
    plt.plot(t,-M/np.sqrt(x**2+y**2))
    plt.xlabel('t'); plt.ylabel('KE(blue) and PE(green) (m=1)'); plt.title('Kineric Energy & Potential Energy')
    plt.figure()
    plt.plot(t,x*vy-y*vx)
    plt.axis([0,T,-3,10])
    plt.xlabel('t'); plt.ylabel('Angular Momentum (m=1)'); plt.title('Angular Momentum');
    return x,y

def main2(M,T,Nt,x0,y0,vx0,vy0):
    
    dt=T/(Nt-1); dt2=dt/2; dt6=dt/6
    x=np.zeros(Nt) ; y=np.zeros(Nt); vx=np.zeros(Nt); vy=np.zeros(Nt);
    x[0]=x0; y[0]=y0; vx[0]=vx0; vy[0]=vy0;

    def D(x,y,vx,vy):
        Dx=vx; Dy=vy;
        r=sqrt(x**2+y**2);
        Dvx=-M*x/r**3; Dvy=-M*y/r**3;
        return Dx,Dy,Dvx,Dvy
    
    s=0; s_last=0; s_last_last=0; # the distance from the beginning
    outgoing=True; # The status that s is increasing.
    iiend=0; # The ending step number
    for ii in range(0,Nt-1):
        k1x,k1y,k1vx,k1vy=D(x[ii],y[ii],vx[ii],vy[ii])
        k2x,k2y,k2vx,k2vy=D(x[ii]+dt2*k1x,y[ii]+dt2*k1y,vx[ii]+dt2*k1vx,vy[ii]+dt2*k1vy)
        k3x,k3y,k3vx,k3vy=D(x[ii]+dt2*k2x,y[ii]+dt2*k2y,vx[ii]+dt2*k2vx,vy[ii]+dt2*k2vy)
        k4x,k4y,k4vx,k4vy=D(x[ii]+dt *k3x,y[ii]+dt *k3y,vx[ii]+dt *k3vx,vy[ii]+dt *k3vy)
        x[ii+1]=x[ii]+dt6*(k1x+2*k2x+2*k3x+k4x)
        y[ii+1]=y[ii]+dt6*(k1y+2*k2y+2*k3y+k4y)
        vx[ii+1]=vx[ii]+dt6*(k1vx+2*k2vx+2*k3vx+k4vx)
        vy[ii+1]=vy[ii]+dt6*(k1vy+2*k2vy+2*k3vy+k4vy)
        s,s_last,s_last_last=sqrt((x[ii]-x0)**2+(y[ii]-y0)**2),s,s_last
        if outgoing & (s<s_last):
            outgoing=False
            a=s_last/2; print('a=',a,'step=',ii-1)
        if (not outgoing) & (s_last_last <= s_last) & (s_last <= s):
            Period=dt*(ii-1); print('Finished! Period=', Period,'step=',ii-1); iiend=ii-1;
            print('Period^2/a^3=',Period**2/a**3); print('')
            break
        
    if iiend==0:
        print('Error! Need larger T!'); return
    else:
        t=np.linspace(0,T,Nt); t=t[:iiend];
        x=x[:iiend]; y=y[:iiend]; vx=vx[:iiend]; vy=vy[:iiend];
        plt.figure()
        plt.plot(t,0.5*(vx**2+vy**2));
        plt.plot(t,-M/np.sqrt(x**2+y**2))
        plt.xlabel('t'); plt.ylabel('KE(blue) and PE(green) (m=1)'); plt.title('Kineric Energy & Potential Energy')
        plt.figure()
        plt.plot(t,x*vy-y*vx)
        plt.axis([0,T,-3,10])
        plt.xlabel('t'); plt.ylabel('Angular Momentum (m=1)'); plt.title('Angular Momentum');
        return x,y

# Two body problem

def main3(M0,m,M,T,Nt,x0,y0,vx0,vy0,X0,Y0,VX0,VY0):
    
    dt=T/(Nt-1); dt2=dt/2; dt6=dt/6
    x=np.zeros(Nt); y=np.zeros(Nt); vx=np.zeros(Nt); vy=np.zeros(Nt);
    X=np.zeros(Nt); Y=np.zeros(Nt); VX=np.zeros(Nt); VY=np.zeros(Nt);
    x[0]=x0; y[0]=y0; vx[0]=vx0; vy[0]=vy0; X[0]=X0; Y[0]=Y0; VX[0]=VX0; VY[0]=VY0;

    def D(x,y,vx,vy,X,Y,VX,VY):
        Dx=vx; Dy=vy; DX=VX; DY=VY
        r=sqrt(x**2+y**2); R=sqrt(X**2+Y**2); rR=sqrt((X-x)**2+(Y-y)**2) #distance between m,M
        Dvx=-M0*x/r**3-M*(x-X)/rR**3;
        Dvy=-M0*y/r**3-M*(y-Y)/rR**3;
        DVX=-M0*X/R**3-m*(X-x)/rR**3;
        DVY=-M0*Y/R**3-m*(Y-y)/rR**3;
        return Dx,Dy,Dvx,Dvy,DX,DY,DVX,DVY
    
    for ii in range(0,Nt-1):
        k1x,k1y,k1vx,k1vy,K1X,K1Y,K1VX,K1VY=D(x[ii],y[ii],vx[ii],vy[ii],X[ii],Y[ii],VX[ii],VY[ii])
        temp=[x[ii]+dt2*k1x,y[ii]+dt2*k1y,vx[ii]+dt2*k1vx,vy[ii]+dt2*k1vy,X[ii]+dt2*K1X,Y[ii]+dt2*K1Y,VX[ii]+dt2*K1VX,VY[ii]+dt2*K1VY]
        k2x,k2y,k2vx,k2vy,K2X,K2Y,K2VX,K2VY=D(*temp)
        temp=[x[ii]+dt2*k2x,y[ii]+dt2*k2y,vx[ii]+dt2*k2vx,vy[ii]+dt2*k2vy,X[ii]+dt2*K2X,Y[ii]+dt2*K2Y,VX[ii]+dt2*K2VX,VY[ii]+dt2*K2VY]
        k3x,k3y,k3vx,k3vy,K3X,K3Y,K3VX,K3VY=D(*temp)
        temp=[x[ii]+dt*k3x,y[ii]+dt*k3y,vx[ii]+dt*k3vx,vy[ii]+dt*k3vy,X[ii]+dt*K3X,Y[ii]+dt*K3Y,VX[ii]+dt2*K3VX,VY[ii]+dt*K3VY]
        k4x,k4y,k4vx,k4vy,K4X,K4Y,K4VX,K4VY=D(*temp)
        x[ii+1]=x[ii]+dt6*(k1x+2*k2x+2*k3x+k4x)
        y[ii+1]=y[ii]+dt6*(k1y+2*k2y+2*k3y+k4y)
        vx[ii+1]=vx[ii]+dt6*(k1vx+2*k2vx+2*k3vx+k4vx)
        vy[ii+1]=vy[ii]+dt6*(k1vy+2*k2vy+2*k3vy+k4vy)
        X[ii+1]=X[ii]+dt6*(K1X+2*K2X+2*K3X+K4X)
        Y[ii+1]=Y[ii]+dt6*(K1Y+2*K2Y+2*K3Y+K4Y)
        VX[ii+1]=VX[ii]+dt6*(K1VX+2*K2VX+2*K3VX+K4VX)
        VY[ii+1]=VY[ii]+dt6*(K1VY+2*K2VY+2*K3VY+K4VY)
    t=np.linspace(0,T,Nt);
    # Energies for m
    plt.figure()
    plt.plot(t,0.5*m*(vx**2+vy**2));
    plt.plot(t,-M0*m/np.sqrt(x**2+y**2)-M*m/np.sqrt((x-X)**2+(y-Y)**2));
    plt.xlabel('t'); plt.ylabel('KE(blue) and PE(green)'); plt.title('KE & PE for m')
    # Energies for M
    plt.figure()
    plt.plot(t,0.5*M*(vx**2+vy**2));
    plt.plot(t,-M0*M/np.sqrt(X**2+Y**2)-m*M/np.sqrt((x-X)**2+(y-Y)**2));
    plt.xlabel('t'); plt.ylabel('KE(blue) and PE(green)'); plt.title('KE & PE for M')
    # Total Energy
    plt.figure()
    plt.plot(t,0.5*m*(vx**2+vy**2)+0.5*M*(VX**2+VY**2)\
             -M0*M/np.sqrt(X**2+Y**2)-M0*m/np.sqrt(x**2+y**2)-m*M/np.sqrt((x-X)**2+(y-Y)**2));
    plt.xlabel('t'); plt.ylabel('E'); plt.title('Total Energy')     
    # Angular Momentum for m
    plt.figure()
    plt.plot(t,m*(x*vy-y*vx))
    plt.axis([0,T,-3,10])
    plt.xlabel('t'); plt.ylabel('Angular Momentum'); plt.title('Angular Momentum for m');
    # Angular Momentum for M
    plt.figure()
    plt.plot(t,M*(X*VY-Y*VX))
    plt.axis([0,T,-3,10])
    plt.xlabel('t'); plt.ylabel('Angular Momentum'); plt.title('Angular Momentum for M');
    # Total Angular Momentum
    plt.figure()
    plt.plot(t,m*(x*vy-y*vx)+M*(X*VY-Y*VX))
    plt.axis([0,T,-3,10])
    plt.xlabel('t'); plt.ylabel('Angular Momentum'); plt.title('Total Angular Momentum');
    
    
    return x,y,X,Y
